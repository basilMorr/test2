function adjustMenuBarWidth() { const lottieElement = document.querySelector('#lottie-container svg'); const menuBar = document.getElementById('menu-bar'); if (lottieElement) { const lottieRect = lottieElement.getBoundingClientRect(); const lottieWidth = lottieRect.width; const lottieLeft = lottieRect.left; console.log(`Lottie Element Width: ${lottieWidth}, Left: ${lottieLeft}`); // Set the menu bar width and position to match the Lottie element menuBar.style.width = `${lottieWidth}px`; menuBar.style.left = `${lottieLeft}px`; console.log(`Adjusted Menu Bar Width to: ${lottieWidth} and Left to: ${lottieLeft}`); } else { console.error('Lottie SVG element not found.'); } }
